Quick Commands

# Restore soul
cd ~/.openclaw
git clone https://github.com/Colimaforever/truth-soul.git workspace

# Restore site
cd ~/.openclaw/workspace
git clone https://github.com/Colimaforever/Methodictruth.git site

# Configure
openclaw setup
openclaw configure --section telegram

# Start
openclaw gateway start

# Switch model
openclaw model set anthropic/claude-opus-4-6
openclaw gateway restart
